    const selectedType = cpuTypeSelect.value;
    cpuModelSelect.innerHTML = "";

      cpuModelSelect.innerHTML = `<option value="">اختر المعالج</option>`;

  document.getElementById("checkForm").addEventListener("submit", (e) => {
    e.preventDefault();
    const cpuType = cpuTypeSelect.value;
    const cpuModel = cpuModelSelect.value;
    const gpuModel = gpuModelSelect.value;

    if (!cpuType) {
      resultDiv.textContent = "يرجى اختيار نوع المعالج.";
      resultDiv.className = "red";
      return;
    }
    if (!cpuModel) {
      resultDiv.textContent = "يرجى اختيار المعالج.";
      resultDiv.className = "red";
      return;

      return;

    const intelSupportedGpusPrefixes = ["Intel HD", "Intel Iris", "Intel UHD", "Intel Iris Plus", "Intel Iris Xe"];
    const amdSupportedModels = processors.AMD; // من القائمة
    const amdSupportedGpusPrefixes = ["AMD Radeon"];
    // NVIDIA غير مدعومة رسمياً

    if (cpuType === "Intel") {
      const genMatch = cpuModel.match(/(\d+)(st|nd|rd|th) Gen/);
      let gen = 0;
      if (genMatch) gen = parseInt(genMatch[1], 10);
      if (!intelSupportedGens.includes(gen)) {
        unsupportedItems.push(`معالج Intel من الجيل ${gen} غير مدعوم`);
      }
      if (!intelSupportedGpusPrefixes.some(prefix => gpuModel.startsWith(prefix))) {
        unsupportedItems.push(`كرت شاشة Intel من نوع "${gpuModel}" غير مدعوم`);
      }
    } else if (cpuType === "AMD") {
      if (!amdSupportedModels.includes(cpuModel)) {
        unsupportedItems.push(`معالج AMD "${cpuModel}" غير مدعوم أو يحتاج لتعديلات خاصة`);
      }
      if (!amdSupportedGpusPrefixes.some(prefix => gpuModel.startsWith(prefix))) {
        unsupportedItems.push(`كرت شاشة AMD من نوع "${gpuModel}" غير مدعوم`);
      }
    } else if (cpuType === "NVIDIA") {
      unsupportedItems.push("نيفيديا غير مدعومة رسمياً في هاكنتوش");
    } else {
      unsupportedItems.push("نوع المعالج غير معروف");
    }

    if (unsupportedItems.length === 0) {
      resultDiv.textContent = "جهازك مدعوم لتشغيل هاكنتوش بشكل جيد مع هذا المعالج وكرت الشاشة.";

      resultDiv.textContent = msg;
      resultDiv.className = "red";
    }
  });
